#include <stdio.h>
#include <stdlib.h>
#include <time.h>
int main(){
    srand((unsigned int)time(0));
    int a[10],i,j,t;
    printf("输出10个随机两位数：\n");
    for(i=0;i<10;i++){
        a[i]=10+rand()%90;
        printf("%d\t",a[i]);
    }
    printf("\n由小到大排列：");
    for(i=8;i+1;i--){
        for(j=0;j<i+1;j++){
            if(a[j]>a[j+1])t=a[j+1],a[j+1]=a[j],a[j]=t;
        }
    }
    printf("\n");
    for(i=0;i<10;i++){
        printf("%d\t",a[i]);
    }
    printf("\n最小数：%d\n最大数：%d\n中位数：%d\n",a[0],a[9],(a[4]+a[5])/2);
    return 0;
}

#include <stdio.h>
#define N 100
int main(){
    int n1[N]={0},n2[N]={0},m,i,k=0;
    i=3,m=5;
    i--;
    for(;k<100;){
        if(i>N-1){
            i-=N;
        }
        if(n1[i]){
            i++;
            continue;
        }
        else{
            n1[i]=1;
            n2[k]=i+1;
        }
        i+=m+1,k++;
    }
    printf("%d",n2[0]);
    for(k=1;k<100;k++)printf("→%d",n2[k]);
}

#include <stdio.h>

    int x1[8][8],x2[8][8],x3[8][8],x4[8][8],x5[8][8],x6[8][8],x7[8][8],x8[8][8],j1,j2,j3,j4,j5,j6,j7,j8,m,n,c=0;//可怕的变量
    //第一个
    for(j1=0;j1<8;j1++){
        for(m=0;m<8;m++)for(n=0;n<8;n++)x1[m][n]=8;

        for(n=0;((n)<8)&&((j1+n)<8);n++)x1[n][j1+n]=0;
        for(n=0;((n)<8)&&((j1-n)>=0);n++)x1[n][j1-n]=0;

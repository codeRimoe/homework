#include <stdio.h>
int main(){
    float r,d,c,s,pi;
    printf("请输入\n圆的半径r=");
    scanf("%f",&r);
    pi=3.14159;
    d=2*r;
    c=pi*d;
    s=pi*r*r;
    printf("圆的直径d=%f\n圆的周长C=%f\n圆的面积S=%f\n",d,c,s);
    return 0;
}

void s(long x){
    long i;

}

#include <stdio.h>
#define N 100
int eng(char c){
    if((c>='A'&&c<='Z')||(c>='a'&&c<='z')) return 1;
    else return 0;
}
int dot(char s[]){
    int p=0;
    for(;s[p]!='\0';p++);
    return p;
}
void longest(char s[]){
    int p=0,i=0,l=0,j=0;
    for(;eng(s[p])==0;p++);//start form nenglish char
    for(;p<=dot(s);p++,i++){
        if(eng(s[p])==0){
            if(i>l)l=i;
            i=-1;//cos i++ start from a non-eng char,we should let the first countless
        }
    }
    for(i=0,p=0;p<=dot(s);p++,i++){
        if(eng(s[p])==0){
            if(i==l){
                for(j=p-i;j!=p;j++)printf("%c",s[j]);
                printf("\t");
            }
            i=-1;
        }
    }
    printf("\n");
}
int main(){
    char s[N];
    printf("Please input a line of word,input \"^Q\"(without \"\") to exit.\n");
in:gets(s);
    if(s[0]=='^'&&s[1]=='Q') goto exit;
    longest(s);
    goto in;
exit:return 0;
}
